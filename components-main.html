Чекбокс

                    
                        <div role="group">
                            <input id="checkbox1" type="checkbox" role="checkbox" name="checkbox1" value="чекбокс1"
                                   aria-checked="false" aria-labelledby="checkboxLabel" disabled>
                            <label id="checkboxLabel" for="checkbox1">чекбокс1 (disabled)</label>
                            <input id="checkbox2" type="checkbox" role="checkbox" name="checkbox2" value="чекбокс2"
                                   aria-labelledby="checkboxLabel">
                            <label id="checkboxLabel" for="checkbox2">чекбокс2 (default)</label>
                            <input id="checkbox2" type="checkbox" role="checkbox" name="checkbox2" value="чекбокс2"
                                   aria-labelledby="checkboxLabel" checked>
                            <label id="checkboxLabel" for="checkbox2">чекбокс3 (checked)</label>
                        </div>

Радио

                    
                        <div role="radiogroup">
                            <input id="radio1" type="radio" name="first" value="Первый">
                            <label id="label1" for="radio1">Первый</label>
        
                            <input id="radio2" type="radio" name="second" value="Второй" disabled>
                            <label id="label2" for="radio2">Второй</label>
        
                            <input id="radio3" type="radio" name="third" value="Третий" checked>
                            <label id="label3" for="radio3">Третий</label>
                        </div>
